Например: не отличает 0 от false или от пустой строки: alert( 0 == false ); //true alert( '' == false ); //true СТРОГИЙ ЖЕ ТИП - не преобразует операнды разных типов в числа. alert( 0 === false ); //false, так как сравниваются разные типы.
